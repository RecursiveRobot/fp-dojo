Impure vs. Pure


// impure
var minimum = 21;

var checkAge = function(age) {
  return age >= minimum;
};


// pure
var checkAge = function(age) {
  var minimum = 21;
  return age >= minimum;
};

Pure or Impure?


var Greeter = (function () {
    function Greeter(message) {
        this.greeting = message;
    }
    Greeter.prototype.greet = function () {
        return "Hello, " + this.greeting;
    };
    return Greeter;
}());

Impure!


var Greeter = (function () {
    function Greeter(message) {
        this.greeting = message;
    }
    Greeter.prototype.greet = function () {
        return "Hello, " + this.greeting;
    };
    return Greeter;
}());

var greeter = new Greeter("world");
greeter.greet();
// "Hello, world"

greeter.greeting = "darkness, my old friend...";
greeter.greet();
// "Hello, darkness, my old friend..."

Function stored in Variable

Function as Parameter


var double = x => x * 2;

function apply(func, value) {
  return func(value);
};

apply(double, 5);
// 10

Function as Result


var getDoubleFunction = function() {
  return x => x * 2;
};

var double = getDoubleFunction();

double(5);
// 10

Addition


// Uncurried..
function add(a, b) {
    return a + b;
}

var c = add(1, 2);
// 3

Addition


// Curried...
function addCurried(a) {
    return function(b) {
        return a + b;
    }
}

var d = addCurried(1)(2);
// 3

var add5 = addCurried(5);
// add5 :: int -> int
var e = add5(2);
// 7

Ramda to the Rescue!


// Ramda
var R = require('ramda');

function addNumbers(a, b, c, d, e){
  return a + b + c + d + e;
}

// curriedAddition :: int -> int -> int -> int -> int -> int
var curriedAddition = R.curry(addNumbers);

var f = curriedAddition(1)(2)(5)(3)(4);
// 15
